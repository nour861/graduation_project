class Special {
  late final int Result;
  late MyData data;
  late final Null Message;

  Special.fromJson(Map<String, dynamic> json) {
    Result = json['Result'];
    data = MyData.fromJson(json['Data']);
    Message = null;
  }
}

class MyData {
  late final int Id;
  late final String Name;

  MyData.fromJson(Map<String, dynamic> json) {
    Id = json['Id'];
    Name = json['Name'];
  }
}
